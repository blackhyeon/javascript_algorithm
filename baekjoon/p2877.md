* 구현
* 시간복잡도 log(n)
* readline 설정이 쉽지 않았다.

```javascript
const readline = require('readline');

function solution(k) {
    numbers = ['4', '7'] // 99  
    let i = k -1;
    let size = numbers.length;
    answer = "";
    while (true) {
        answer = numbers[i % size] + answer;
        if (i < size) break;
        i = parseInt(i / size) - 1;
    }
    return answer;
}

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

rl.on('line', function(line) {
    console.log(solution(parseInt(line)));
    rl.close();
})
rl.on('close', function(line) {
    process.exit();
})

```
